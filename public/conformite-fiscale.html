Tables Supabase requises
-- Statut des envois PDP CREATE TABLE invoice_pdp_status ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), invoice_id UUID REFERENCES invoices(id) ON DELETE CASCADE, status VARCHAR(20) CHECK (status IN ('sent','accepted','rejected','pending')), response_payload JSONB, error_message TEXT, created_at TIMESTAMP DEFAULT NOW() ); -- Historique des echanges PDP CREATE TABLE pdp_exchange_logs ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), invoice_id UUID REFERENCES invoices(id) ON DELETE SET NULL, request_payload JSONB NOT NULL, response_payload JSONB, http_status INT, duration_ms INT, created_at TIMESTAMP DEFAULT NOW() ); -- E-reporting batches (B2C) CREATE TABLE ereporting_batches ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), period DATE NOT NULL, generated_at TIMESTAMP DEFAULT NOW(), file_path TEXT, file_format VARCHAR(10) DEFAULT 'csv', status VARCHAR(20) DEFAULT 'generated', total_amount DECIMAL(12,2), transaction_count INT ); -- Index de performance CREATE INDEX idx_pdp_status_invoice ON invoice_pdp_status(invoice_id); CREATE INDEX idx_pdp_logs_invoice ON pdp_exchange_logs(invoice_id); CREATE INDEX idx_pdp_logs_date ON pdp_exchange_logs(created_at DESC);
